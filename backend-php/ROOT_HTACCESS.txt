# If backend-php is in a subdirectory, you may need to place this .htaccess
# in your website ROOT directory instead of backend-php/

# Enable Rewrite Engine
RewriteEngine On

# Route /api requests to backend-php/index.php
RewriteCond %{REQUEST_URI} ^/api/(.*)$
RewriteRule ^api/(.*)$ backend-php/index.php?route=$1 [QSA,L]

# Allow CORS
Header set Access-Control-Allow-Origin "*"
Header set Access-Control-Allow-Methods "GET, POST, PUT, DELETE, PATCH, OPTIONS"
Header set Access-Control-Allow-Headers "Content-Type, Authorization"

# Handle preflight requests
RewriteCond %{REQUEST_METHOD} OPTIONS
RewriteRule ^(.*)$ $1 [R=204,L]
